# to run psql in docker
 - docker exec -it postgres-db psql -U user -d learn-6
 # database:
  - \dt => show all tables
  - \d <table_name> => show table schema
  - \c <database_name> => connect to database
  - \q => quit
  - \x (on/off) => expand table
  - \l => show all databases
  - \dn => show all schemas
  - \dn+ => show all schemas with details
